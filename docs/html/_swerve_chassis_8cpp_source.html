98 void SwerveChassis::Drive( units::meters_per_second_t xSpeed,
99  units::meters_per_second_t ySpeed,
100  units::radians_per_second_t rot,
101  bool fieldRelative)
102 {
103  units::degree_t yaw{m_pigeon->GetYaw()};
104  Rotation2d currentOrientation {yaw};
105  auto states = m_kinematics.ToSwerveModuleStates(
106  fieldRelative ? frc::ChassisSpeeds::FromFieldRelativeSpeeds(
107  xSpeed, ySpeed, rot, currentOrientation) : frc::ChassisSpeeds{xSpeed, ySpeed, rot} );
108 
109  m_kinematics.NormalizeWheelSpeeds(&states, m_maxSpeed);
110 
111  auto [fl, fr, bl, br] = states;
112 
113  m_frontLeft.get()->SetDesiredState(fl);
114  m_frontRight.get()->SetDesiredState(fr);
115  m_backLeft.get()->SetDesiredState(bl);
116  m_backRight.get()->SetDesiredState(br);
117 }

134 void SwerveChassis::Drive( double drive, double steer, double rotate, bool fieldRelative )
135 {
136  // scale joystick values to velocities using max chassis values
137  auto maxSpeed = GetMaxSpeed();
138  auto maxRotation = GetMaxAngularSpeed();
139 
140  units::velocity::meters_per_second_t driveSpeed = drive * maxSpeed;
141  units::velocity::meters_per_second_t steerSpeed = steer * maxSpeed;
142  units::angular_velocity::radians_per_second_t rotateSpeed = rotate * maxRotation;
143 
144  Drive( driveSpeed, steerSpeed, rotateSpeed, fieldRelative );
145 }

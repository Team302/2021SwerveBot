67 void SwerveDrive::Run( )
68 {
69  double drive = 0.0;
70  double steer = 0.0;
71  double rotate = 0.0;
72  auto controller = GetController();
73  if ( controller != nullptr )
74  {
75  drive = controller->GetAxisValue( TeleopControl::FUNCTION_IDENTIFIER::SWERVE_DRIVE_DRIVE) ;
76  steer = controller->GetAxisValue( TeleopControl::FUNCTION_IDENTIFIER::SWERVE_DRIVE_STEER);
77  rotate = controller->GetAxisValue( TeleopControl::FUNCTION_IDENTIFIER::SWERVE_DRIVE_ROTATE);
78  }
79 
80  m_chassis.get()->Drive(drive, steer, rotate, true);
81  return;
82 
83  //Get drive, steer, and rotate values
84  units::velocity::meters_per_second_t maxSpeed = m_chassis.get()->GetMaxSpeed();
85  units::velocity::meters_per_second_t driveSpeed = maxSpeed * drive;
86  units::velocity::meters_per_second_t turnSpeed = maxSpeed * steer;
87 
88  units::radians_per_second_t maxRotateSpeed = m_chassis.get()->GetMaxAngularSpeed();
89  units::radians_per_second_t rotateSpeed = maxRotateSpeed * rotate;
90 
91  m_chassis.get()->Drive(driveSpeed, turnSpeed, rotateSpeed, true);
92 }

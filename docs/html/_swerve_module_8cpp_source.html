183 SwerveModuleState SwerveModule::GetState() const
184 {
185  // Get the Current Chassis Velocity
186  auto chassis = SwerveChassisFactory::GetSwerveChassisFactory()->GetSwerveChassis();
187  units::velocity::meters_per_second_t mps = units::length::meter_t(GetWheelDiameter() * pi ) / units::angle::radian_t(1.0) * units::angular_velocity::radians_per_second_t(m_driveMotor->GetRPS()*2.0*pi);

191 
192  // Create the state and return it

194  return state;

201 void SwerveModule::SetDesiredState
202 (
203  const SwerveModuleState& referenceState
204 )
205 {
206  // Update targets so the angle turned is less than 90 degrees
207  // If the desired angle is less than 90 degrees from the target angle (e.g., -90 to 90 is the amount of turn), just use the angle and speed values
208  // if it is more than 90 degrees (90 to 270), the can turn the opposite direction -- increase the angle by 180 degrees -- and negate the wheel speed
209  // finally, get the value between -90 and 90
210  auto state = SwerveModuleState::Optimize(referenceState, units::angle::degree_t(m_turnSensor.get()->GetAbsolutePosition()));
211 
212  // Set Drive Target
213  // convert mps to unitless rps by taking the speed and dividing by the circumference of the wheel
214  auto driveTarget = state.speed.to<double>() / units::length::meter_t(m_wheelDiameter).to<double>() * wpi::math::pi;
215 
216  // Set Turn Target
217  auto turnTarget = state.angle.Degrees().to<double>();
218  auto targetCounts = ConvertTargetAngleToCounts( turnTarget );
219 
220  // Run the motors
221  m_nt.get()->PutNumber( "drive target", driveTarget );
222  m_nt.get()->PutNumber( "target counts", targetCounts );
223 
224  m_driveMotor.get()->Set(m_nt, driveTarget);
225  m_turnMotor.get()->Set(m_nt, targetCounts);
226  //m_turnMotor.get()->Set(m_nt, turnTarget);
227 }

229 double SwerveModule::ConvertTargetAngleToCounts
230 (
231  double targetAngle
232 )
233 {
234  m_nt.get()->PutNumber( "current angle", m_turnSensor.get()->GetAbsolutePosition() );
235  m_nt.get()->PutNumber( "target angle", targetAngle );
236 
237  auto deltaDegs = m_initialAngle - targetAngle;
238  //auto deltaCounts = deltaDegs * 8.5333 * 2048 / 360.0;
239  auto deltaCounts = (deltaDegs / 360.0) * 2048;
240  auto targetCounts = m_initialCounts + deltaCounts;
241 
242  m_nt.get()->PutNumber( "initial angle", m_initialAngle );
243  m_nt.get()->PutNumber( "delta degrees", deltaDegs );
244  m_nt.get()->PutNumber( "delta counts", deltaCounts );
245  m_nt.get()->PutNumber( "initial counts", m_initialCounts );
246  m_nt.get()->PutNumber( "target counts", targetCounts );
247 
248  return targetCounts;
249 }
